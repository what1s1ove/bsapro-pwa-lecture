Service Worker

  • HTTPS only
  • New thread only (background mode)
  • No DOM
  • No sync API (XHR, LocalStorage, etc.)
  • Requests interception (caching, mutation requests)
  • this self

Service Worker Life Cycle

      // sw.js
      

        self.addEventListener('install', () => {
            // add source to cache
          });
      

      

         
        self.addEventListener('activate', () => {
          // version control
        });
      

      

         
        self.addEventListener('fetch', () => {
          // requests interception (response with resources from cache)
        });

Service Worker Example

      // sw.js
      

        const APP_CACHE_NAME = 'app-cache-v1';
        const ASSETS_URLS = ['index.html', '/js/app.js', '/css/main.css'];
      

      

         
        self.addEventListener('install', async () => {
          const cache = await caches.open(APP_CACHE_NAME);
          await cache.addAll(ASSETS_URLS);
        });
      

      

         
        self.addEventListener('fetch', async (evt) => {
          const cached = await caches.match(evt.request);
          evt.respondWith(cached ?? await fetch(evt.request));
        });
      

      

         
        // js/main.js
        window.addEventListener('load', () => { 
          navigator.serviceWorker?.register('../sw.js'); 
        });
